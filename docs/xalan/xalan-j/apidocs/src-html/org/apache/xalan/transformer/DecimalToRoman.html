021    package org.apache.xalan.transformer;
022    
023    /**
024     * Structure to help in converting integers to roman numerals
025     * @xsl.usage internal
026     */
027    public class DecimalToRoman
028    {
029    
030      /**
031       * Constructor DecimalToRoman
032       *
033       *
034       * @param postValue Minimum value for a given range of 
035       * roman numbers
036       * @param postLetter Correspoding letter (roman) to postValue
037       * @param preValue Value of last prefixed number within 
038       * that same range (i.e. IV if postval is 5 (V))
039       * @param preLetter Correspoding letter(roman) to preValue
040       */
041      public DecimalToRoman(long postValue, String postLetter, long preValue,
042                            String preLetter)
043      {
044    
045        this.m_postValue = postValue;
046        this.m_postLetter = postLetter;
047        this.m_preValue = preValue;
048        this.m_preLetter = preLetter;
049      }
050    
051      /** Minimum value for a given range of roman numbers          */
052      public long m_postValue;
053    
054      /** Correspoding letter (roman) to m_postValue          */
055      public String m_postLetter;
056    
057      /** Value of last prefixed number within that same range  */
058      public long m_preValue;
059    
060      /** Correspoding letter (roman) to m_preValue          */
061      public String m_preLetter;
062    }
