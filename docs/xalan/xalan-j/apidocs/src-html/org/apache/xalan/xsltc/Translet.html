028     * @author Jacek Ambroziak

031    public interface Translet {
032    
033        public void transform(DOM document, SerializationHandler handler)
034            throws TransletException;
035        public void transform(DOM document, SerializationHandler[] handlers)
036            throws TransletException;

038                              SerializationHandler handler)
039            throws TransletException;
040    
041        public Object addParameter(String name, Object value);
042    
043        public void buildKeys(DOM document, DTMAxisIterator iterator,
044                              SerializationHandler handler, int root)
045            throws TransletException;
046        public void addAuxiliaryClass(Class auxClass);
047        public Class getAuxiliaryClass(String className);
048        public String[] getNamesArray();
049        public String[] getUrisArray();
050        public int[]    getTypesArray();
051        public String[] getNamespaceArray();
052    }
