021    package org.apache.xml.utils;
022    
023    /**
024     * An interface that a Serializer/ContentHandler/ContentHandler must
025     * implement in order for disable-output-escaping to work.
026     * @xsl.usage advanced
027     */
028    public interface RawCharacterHandler
029    {
030    
031      /**
032       * Serialize the characters without escaping.
033       *
034       * @param ch Array of characters
035       * @param start Start index of characters in the array
036       * @param length Number of characters in the array
037       *
038       * @throws javax.xml.transform.TransformerException
039       */
040      public void charactersRaw(char ch[], int start, int length)
041        throws javax.xml.transform.TransformerException;
042    }
