00729                 XalanUnicode::charGreaterThanSign == chars[i + 2])
00730             {
00731                 if (outsideCDATA == true)
00732                 {
00733                     m_writer.write(
00734                         m_constants.s_cdataCloseString,
00735                         m_constants.s_cdataCloseStringLength);
00736                 }
00737 
00738                 m_writer.write(value_type(XalanUnicode::charRightSquareBracket));
00739                 m_writer.write(value_type(XalanUnicode::charRightSquareBracket));

00745                 m_writer.write(
