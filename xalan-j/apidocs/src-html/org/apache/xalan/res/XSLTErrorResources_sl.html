041    /*
042     * This file contains error and warning messages related to Xalan Error
043     * Handling.
044     *
045     *  General notes to translators:
046     *
047     *  1) Xalan (or more properly, Xalan-interpretive) and XSLTC are names of
048     *     components.
049     *     XSLT is an acronym for "XML Stylesheet Language: Transformations".
050     *     XSLTC is an acronym for XSLT Compiler.
051     *
052     *  2) A stylesheet is a description of how to transform an input XML document
053     *     into a resultant XML document (or HTML document or text).  The
054     *     stylesheet itself is described in the form of an XML document.
055     *
056     *  3) A template is a component of a stylesheet that is used to match a
057     *     particular portion of an input document and specifies the form of the
058     *     corresponding portion of the output document.
059     *
060     *  4) An element is a mark-up tag in an XML document; an attribute is a
061     *     modifier on the tag.  For example, in <elem attr='val' attr2='val2'>
062     *     "elem" is an element name, "attr" and "attr2" are attribute names with
063     *     the values "val" and "val2", respectively.
064     *
065     *  5) A namespace declaration is a special attribute that is used to associate
066     *     a prefix with a URI (the namespace).  The meanings of element names and
067     *     attribute names that use that prefix are defined with respect to that
068     *     namespace.
069     *
070     *  6) "Translet" is an invented term that describes the class file that
071     *     results from compiling an XML stylesheet into a Java class.
072     *
073     *  7) XPath is a specification that describes a notation for identifying
074     *     nodes in a tree-structured representation of an XML document.  An
075     *     instance of that notation is referred to as an XPath expression.
076     *
077     */

1453      /**
1454       *   Return a named ResourceBundle for a particular locale.  This method mimics the behavior
1455       *   of ResourceBundle.getBundle().
1456       *
1457       *   @param className the name of the class that implements the resource bundle.
1458       *   @return the ResourceBundle
1459       *   @throws MissingResourceException
1460       */
1461      public static final XSLTErrorResources loadResourceBundle(String className)
1462              throws MissingResourceException
1463      {
1464    
1465        Locale locale = Locale.getDefault();
1466        String suffix = getResourceSuffix(locale);
1467    
1468        try
1469        {
1470    
1471          // first try with the given locale
1472          return (XSLTErrorResources) ResourceBundle.getBundle(className
1473                  + suffix, locale);
1474        }
1475        catch (MissingResourceException e)
1476        {
1477          try  // try to fall back to en_US if we can't load
1478          {
1479    
1480            // Since we can't find the localized property file,
1481            // fall back to en_US.
1482            return (XSLTErrorResources) ResourceBundle.getBundle(className,
1483                    new Locale("sl", "US"));
1484          }
1485          catch (MissingResourceException e2)
1486          {
1487    
1488            // Now we are really in trouble.
1489            // very bad, definitely very bad...not going to get very far
1490            throw new MissingResourceException(
1491              "Could not load any resource bundles.", className, "");
1492          }
1493        }
1494      }

1496      /**
1497       * Return the resource file suffic for the indicated locale
1498       * For most locales, this will be based the language code.  However
1499       * for Chinese, we do distinguish between Taiwan and PRC
1500       *
1501       * @param locale the locale
1502       * @return an String suffix which canbe appended to a resource name
1503       */
1504      private static final String getResourceSuffix(Locale locale)
1505      {
1506    
1507        String suffix = "_" + locale.getLanguage();
1508        String country = locale.getCountry();
1509    
1510        if (country.equals("TW"))
1511          suffix += "_" + country;
1512    
1513        return suffix;
1514      }
